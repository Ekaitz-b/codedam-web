/** * Gestiona las operaciones de una cuenta bancaria. * * @author Tu Nombre * @version 1.0 */ public class CuentaBancaria { private String titular; private double saldo; /** * Crea una cuenta bancaria con saldo inicial. * * @param titular nombre del titular de la cuenta * @param saldoInicial saldo inicial (debe ser >= 0) */ public CuentaBancaria(String titular, double saldoInicial) { this.titular = titular; this.saldo = saldoInicial; } /** * Deposita dinero en la cuenta. * * @param cantidad cantidad a depositar (debe ser > 0) * @throws IllegalArgumentException si cantidad <= 0 */ public void depositar(double cantidad) { if (cantidad <= 0) { throw new IllegalArgumentException("Cantidad debe ser positiva"); } saldo += cantidad; } /** * Retira dinero de la cuenta. * * @param cantidad cantidad a retirar * @return true si se pudo retirar, false si saldo insuficiente */ public boolean retirar(double cantidad) { if (cantidad > saldo) { return false; } saldo -= cantidad; return true; } /** * Obtiene el saldo actual. * * @return el saldo de la cuenta */ public double getSaldo() { return saldo; } }
