    Desarrollo Android con Kotlin

    Kotlin es el lenguaje oficial para desarrollo Android, moderno y conciso.

    Estructura de un Proyecto Android

    • app/manifests/: AndroidManifest.xml (configuración de la app)
    • app/java/: Código fuente Kotlin/Java
    • app/res/: Recursos (layouts, imágenes, strings)
    • build.gradle: Configuración de compilación

    Activity Principal

    // MainActivity.kt package com.codedam.miapp import android.os.Bundle import androidx.appcompat.app.AppCompatActivity import android.widget.Button import android.widget.TextView import android.widget.Toast class MainActivity : AppCompatActivity() { private lateinit var textView: TextView private lateinit var button: Button override fun onCreate(savedInstanceState: Bundle?) { super.onCreate(savedInstanceState) setContentView(R.layout.activity_main) // Inicializar vistas textView = findViewById(R.id.textView) button = findViewById(R.id.button) // Evento click button.setOnClickListener { textView.text = "¡Hola CodeDAM!"     SharedPreferences

    Room Database

    // Entidad @Entity(tableName = "productos") data class Producto( @PrimaryKey(autoGenerate = true) val id: Int = 0, @ColumnInfo(name = "nombre") val nombre: String, @ColumnInfo(name = "precio") val precio: Double ) // DAO @Dao interface ProductoDao { @Query("SELECT * FROM productos") fun getAll(): List<Producto> @Insert fun insert(producto: Producto) @Delete fun delete(producto: Producto) } // Database @Database(entities = [Producto::class], version = 1) abstract class AppDatabase : RoomDatabase() { abstract fun productoDao(): ProductoDao } // Uso val db = Room.databaseBuilder( applicationContext, AppDatabase::class.java, "mi-database" ).build() val productos = db.productoDao().getAll()

    Retrofit - Consumir APIs REST

    // Interfaz API interface ApiService { @GET("productos") suspend fun getProductos(): List<Producto> @GET("productos/{id}") suspend fun getProducto(@Path("id") id: Int): Producto @POST("productos") suspend fun crearProducto(@Body producto: Producto): Producto } // Cliente Retrofit val retrofit = Retrofit.Builder() .baseUrl("https://api.ejemplo.com/") .addConverterFactory(GsonConverterFactory.create()) .build() val apiService = retrofit.create(ApiService::class.java) // Uso con Coroutines lifecycleScope.launch { try { val productos = apiService.getProductos() // Actualizar UI } catch (e: Exception) { Toast.makeText(this@MainActivity, "Error: ${e.message}", Toast.LENGTH_SHORT).show() } }
