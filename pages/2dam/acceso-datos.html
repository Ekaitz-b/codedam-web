Acceso a Datos

JDBC, Hibernate, MongoDB y Persistencia

Índice de Contenidos

    JDBC - Java Database Connectivity

    JDBC es la API estándar de Java para conectarse a bases de datos relacionales.

    Conexión a Base de Datos

    import java.sql.*; public class ConexionBD { private static final String URL = "jdbc:mysql://localhost:3306/mi_bd"; private static final String USER = "root"; private static final String PASS = "password"; public static Connection getConnection() throws SQLException { return DriverManager.getConnection(URL, USER, PASS); } public static void main(String[] args) { try (Connection conn = getConnection()) { System.out.println("Conexión exitosa!"); } catch (SQLException e) { e.printStackTrace(); } } }

    Operaciones CRUD con JDBC

    // CREATE - Insertar public void insertarCliente(String nombre, String email) { String sql = "INSERT INTO clientes (nombre, email) VALUES (?, ?)"; try (Connection conn = getConnection(); PreparedStatement stmt = conn.prepareStatement(sql)) { stmt.setString(1, nombre); stmt.setString(2, email); int rows = stmt.executeUpdate(); System.out.println("Filas insertadas: " + rows); } catch (SQLException e) { e.printStackTrace(); } } // READ - Consultar public List<Cliente> obtenerClientes() { List<Cliente> clientes = new ArrayList<>(); String sql = "SELECT * FROM clientes"; try (Connection conn = getConnection(); Statement stmt = conn.createStatement(); ResultSet rs = stmt.executeQuery(sql)) { while (rs.next()) { Cliente c = new Cliente(); c.setId(rs.getInt("id")); c.setNombre(rs.getString("nombre")); c.setEmail(rs.getString("email")); clientes.add(c); } } catch (SQLException e) { e.printStackTrace(); } return clientes; } // UPDATE - Actualizar public void actualizarEmail(int id, String nuevoEmail) { String sql = "UPDATE clientes SET email = ? WHERE id = ?"; try (Connection conn = getConnection(); PreparedStatement stmt = conn.prepareStatement(sql)) { stmt.setString(1, nuevoEmail); stmt.setInt(2, id); stmt.executeUpdate(); } catch (SQLException e) { e.printStackTrace(); } } // DELETE - Eliminar public void eliminarCliente(int id) { String sql = "DELETE FROM clientes WHERE id = ?"; try (Connection conn = getConnection(); PreparedStatement stmt = conn.prepareStatement(sql)) { stmt.setInt(1, id); stmt.executeUpdate(); } catch (SQLException e) { e.printStackTrace(); } }

    Hibernate - ORM (Object-Relational Mapping)

    Hibernate mapea objetos Java a tablas de bases de datos, simplificando el acceso a datos.

    Entidad Hibernate

    import javax.persistence.*; @Entity @Table(name = "productos") public class Producto { @Id @GeneratedValue(strategy = GenerationType.IDENTITY) private Long id; @Column(name = "nombre", nullable = false) private String nombre; @Column(name = "precio") private Double precio; @Column(name = "stock") private Integer stock; // Relación Many-to-One @ManyToOne @JoinColumn(name = "categoria_id") private Categoria categoria; // Getters y Setters public Long getId() { return id; } public void setId(Long id) { this.id = id; } public String getNombre() { return nombre; } public void setNombre(String nombre) { this.nombre = nombre; } // ... }

    Operaciones con Hibernate

    import org.hibernate.Session; import org.hibernate.SessionFactory; import org.hibernate.cfg.Configuration; public class ProductoDAO { private SessionFactory factory; public ProductoDAO() { factory = new Configuration() .configure("hibernate.cfg.xml") .addAnnotatedClass(Producto.class) .buildSessionFactory(); } // CREATE public void guardar(Producto producto) { Session session = factory.getCurrentSession(); try { session.beginTransaction(); session.save(producto); session.getTransaction().commit(); } finally { session.close(); } } // READ public Producto obtener(Long id) { Session session = factory.getCurrentSession(); try { session.beginTransaction(); Producto p = session.get(Producto.class, id); session.getTransaction().commit(); return p; } finally { session.close(); } } // UPDATE public void actualizar(Producto producto) { Session session = factory.getCurrentSession(); try { session.beginTransaction(); session.update(producto); session.getTransaction().commit(); } finally { session.close(); } } // DELETE public void eliminar(Long id) { Session session = factory.getCurrentSession(); try { session.beginTransaction(); Producto p = session.get(Producto.class, id); session.delete(p); session.getTransaction().commit(); } finally { session.close(); } } // Query HQL public List<Producto> buscarPorNombre(String nombre) { Session session = factory.getCurrentSession(); try { session.beginTransaction(); List<Producto> productos = session .createQuery("from Producto where nombre like :nombre") .setParameter("nombre", "%" + nombre + "%") .getResultList(); session.getTransaction().commit(); return productos; } finally { session.close(); } } }

    JPA - Java Persistence API

    JPA es el estándar de Java para ORM. Hibernate es su implementación más popular.

    EntityManager

    import javax.persistence.*; public class ProductoRepository { private EntityManagerFactory emf; public ProductoRepository() { emf = Persistence.createEntityManagerFactory("mi-unidad"); } public void guardar(Producto producto) { EntityManager em = emf.createEntityManager(); em.getTransaction().begin(); em.persist(producto); em.getTransaction().commit(); em.close(); } public Producto buscarPorId(Long id) { EntityManager em = emf.createEntityManager(); Producto p = em.find(Producto.class, id); em.close(); return p; } public List<Producto> buscarTodos() { EntityManager em = emf.createEntityManager(); TypedQuery<Producto> query = em.createQuery("SELECT p FROM Producto p", Producto.class); List<Producto> productos = query.getResultList(); em.close(); return productos; } }

    MongoDB - Base de Datos NoSQL

    MongoDB es una base de datos orientada a documentos JSON, ideal para datos no estructurados.

    Conexión y Operaciones MongoDB en Java

    import com.mongodb.client.*; import org.bson.Document; public class MongoDBExample { private MongoClient mongoClient; private MongoDatabase database; private MongoCollection<Document> collection; public MongoDBExample() { mongoClient = MongoClients.create("mongodb://localhost:27017"); database = mongoClient.getDatabase("tienda"); collection = database.getCollection("productos"); } // Insertar documento public void insertar() { Document doc = new Document("nombre", "Laptop") .append("precio", 899.99) .append("stock", 15) .append("categoria", "Informática"); collection.insertOne(doc); } // Buscar documentos public void buscar() { Document query = new Document("categoria", "Informática"); FindIterable<Document> docs = collection.find(query); for (Document doc : docs) { System.out.println(doc.toJson()); } } // Actualizar public void actualizar() { Document query = new Document("nombre", "Laptop"); Document update = new Document("$set", new Document("precio", 849.99)); collection.updateOne(query, update); } // Eliminar public void eliminar() { Document query = new Document("stock", 0); collection.deleteMany(query); } }

    Ficheros - Manejo de Archivos

    Lectura y Escritura de Ficheros

    import java.io.*; import java.nio.file.*; // Escribir archivo de texto public void escribirTexto(String ruta, String contenido) { try { Files.writeString(Paths.get(ruta), contenido); } catch (IOException e) { e.printStackTrace(); } } // Leer archivo de texto public String leerTexto(String ruta) { try { return Files.readString(Paths.get(ruta)); } catch (IOException e) { e.printStackTrace(); return null; } } // Serialización de objetos public void guardarObjeto(Producto p, String archivo) { try (ObjectOutputStream oos = new ObjectOutputStream(new FileOutputStream(archivo))) { oos.writeObject(p); } catch (IOException e) { e.printStackTrace(); } } // Deserialización public Producto leerObjeto(String archivo) { try (ObjectInputStream ois = new ObjectInputStream(new FileInputStream(archivo))) { return (Producto) ois.readObject(); } catch (IOException | ClassNotFoundException e) { e.printStackTrace(); return null; } }
